23 #define DOF_X 0 // x-displacement

28 #define DOF_W 5 // z-rotation
29 #define DOF_P 6 // fluid pressure
30 #define DOF_RU 7 // rigid x-rotation
31 #define DOF_RV 8 // rigid y-rotation
32 #define DOF_RW 9 // rigid z-rotation
33 #define DOF_T 10 // temperature
34 #define DOF_C 11 // solute concentration

36 // The rotational degrees of freedom are only used for rigid nodes and shells.
37 // The fluid pressure is only used for poroelastic problems.
38 // The rigid rotational degrees of freedom are only used for rigid nodes and only during the creation of the stiffenss matrix
39 // The temperature is only used during heat-conduction problems
40 // The solute concentration is only used in solute transport problems.
