19 //
20 struct FE_BOUNDING_BOX
21 {
22  vec3d r0, r1; // coordinates of opposite corners
23 
24  // center of box
25  vec3d center() { return (r0+r1)*0.5; }
26 
27  // dimensions of box
28  double width () { return (r1.x-r0.x); }
29  double height() { return (r1.y-r0.y); }
30  double depth () { return (r1.z-r0.z); }
31 
32  // max dimension
33  double radius()
34  {
35  double w = width();
36  double h = height();
37  double d = depth();
38 
39  if ((w>=d) && (w>=h)) return w;
40  if ((h>=w) && (h>=d)) return h;
41 
42  return d;
43  }
44 
45  void operator += (vec3d r)
46  {
47  if (r.x < r0.x) r0.x = r.x;
48  if (r.y < r0.y) r0.y = r.y;
49  if (r.z < r0.z) r0.z = r.z;
50  if (r.x > r1.x) r1.x = r.x;
51  if (r.y > r1.y) r1.y = r.y;
52  if (r.z > r1.z) r1.z = r.z;
53  }
54 
55  void inflate(double dx, double dy, double dz)
56  {
57  r0.x -= dx; r1.x += dx;
58  r0.y -= dy; r1.y += dy;
59  r0.z -= dz; r1.z += dz;
60  }
61 
62  // check whether a point is inside or not
63  bool IsInside(vec3d r)
64  {
65  return ((r.x>=r0.x) && (r.y>=r0.y) && (r.z>=r0.z) && (r.x<=r1.x) && (r.y<=r1.y) && (r.z<=r1.z));
66  }
67 };
